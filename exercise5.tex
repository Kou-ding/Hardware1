    add x6, x5, x4
    sub x7, x3, x2
    sll x8, x7, x5
    slt x9, x4, x8
    xor x10, x8, x2
    and x11, x10, x8
    srl x12, x11, x9
    or x13, x12, x3
    sra x14, x4, x9
    sw x11, 0(x5)
    lw x15, 0(x5)
    andi x16, x8, -45
    ori x17, x16, 22
    srli x18, x13, 1
    beq x15, x11, 16
